        import time

        def f(lst, v):
            total = 0
            t1 = time.time()
            for i in lst:
                total += i ** v
            t2 = time.time()
            return total, t2-t1

        def g(lst, v):
            total = 0
            t1 = time.perf_counter_ns()
            for i in lst:
                total += i ** v
            t2 = time.perf_counter_ns()
            return total, (t2-t1)/(10**9)
